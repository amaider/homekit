## initialize:
```c
/* 0 way */
i2c_init(I2C_BUS, I2C_SCL_PIN, I2C_SDA_PIN, I2C_FREQ_400K);
ssd1306_init(&oled);

/* 1st way */
i2c_init(I2C_BUS, I2C_SCL_PIN, I2C_SDA_PIN, I2C_FREQ_400K);
if (ssd1306_init(&oled)) {
    debug("Failed to initialize OLED\n");
    return;
}

/* 2nd way */
void display_init() {
    i2c_init(I2C_BUS, I2C_SCL_PIN, I2C_SDA_PIN, I2C_FREQ_400K);
    if(ssd1306_init(&oled)) {
        debug("Failed to initialize OLED\n");
        return;
    }
    ssd1306_set_whole_display_lighting(&oled, false);
    ssd1306_set_scan_direction_fwd(&oled, false);
}


display_init();
```
